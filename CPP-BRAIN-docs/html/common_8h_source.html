92 std::vector<std::string> CommonComponents::split(const char* str, const char* delimiter)
93 {
94 std::vector<std::string> v1;
95 if (str == nullptr || delimiter == nullptr)
96 {
97 return v1;
98 }
99 // String length is 0
100 if ((*str) == 0 || (*delimiter) == 0)
101 {
102 return v1;
103 }
104
105 std::string t1 = str;
106 int delimiterLen = (int)strlen(delimiter);
107 int pos = 0;
108 int lastPos = 0;
109 while (true)
110 {
111 pos = (int)t1.find(delimiter, pos);
112 if (pos == std::string::npos)
113 {
114 // The last slice
115 int len1 = (int)t1.size() - lastPos;
116 if (len1 >= 0)
117 {
118 std::string t2 = t1.substr(lastPos);
119 v1.push_back(t2);
120 }
121 break;
122 }
123
124 int len1 = pos - lastPos;
125 if (len1 >= 0)
126 {
127 std::string t2 = t1.substr(lastPos, len1);
128 v1.push_back(t2);
129 }
130
131 pos += delimiterLen;
132 lastPos = pos;
133 }
134
135 return v1;
